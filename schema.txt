Users(
	user_id PRIMARY KEY,
	username ,
	name ,
	email,
	password_hash,
	location_id FOREIGN KEY REFERENCES Location(location_id)
)

Authors(
	author_id PRIMARY KEY,
	name,
	email
)

Location(
	location_id PRIMARY KEY,
	street_address,
	postal_code,
	city
)

Publisher(
	publisher_id PRIMARY KEY
	name,
	location_id FOREIGN KEY REFERENCES Location(location_id)
)

Books(
	isbn PRIMARY KEY,
	name,
	edition,
	language,
	genre,
	price,
	quantity_available
)

Follows(
	user_id FOREIGN_KEY REFERENCES users(user_id),
	follows_whom FOREIGN_KEY REFERENCES users(user_id),
	PRIMARY KEY(user_id, follows_whom)
)

Author_of_book(
	isbn FOREIGN KEY REFERENCES Books(isbn),
	author_id FOREIGN KEY REFERENCES Authors(author_id),
	publisher_id FOREIGN KEY REFERENCES Publications(publication_id)
	PRIMARY KEY(isbn, author_id, publication_id)
)

Transaction(
	Tx_id PRIMARY KEY,
	user_id FOREIGN KEY REFERENCES Users(user_id),
	location_id FOREIGN KEY REFERENCES Locations(location_id),
	payment_id FOREIGN KEY REFERENCES Payment(payment_id)
)

books_in_transaction(
    Tx-id FOREIGN KEY REFERENCES Transaction(Tx_id),
    isbn FOREIGN KEY REFERENCES Books(isbn)
)

Payment(
	payment_id PRIMARY KEY,
	price,
	mobile_no,
	payment_type
)

Reviews(
	review_id PRIMARY KEY,
	user_id FOREIGN KEY REFERENCES Users(user_id),
	rating,
	downvotes,
	upvotes
)

Cart(
	cart_id PRIMARY KEY,
	user_id FOREIGN KEY
)

CartBook(
	cart_id FOREIGN KEY
	book_id FOREIGN KEY
)
